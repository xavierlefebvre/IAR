\begin{algorithm}[!ht]
\caption{Evolutionary Algorithm}
\begin{algorithmic}
\Procedure{Loop}{}
\State $N \gets \{n_1,n_2,...,n_{10}\}$ \Comment{random controllers ($n$ nodes, $2n$ connections)}
\State $C \gets \{c_1,c_2,...,c_{15}\}$ \Comment{initial conditions}
\State $E \gets \{e_1,e_2,...,e_4\}$ \Comment{environments}
\State $conditionsNb \gets$ length($C$) $*$ length($E$)
\State $startTime \gets time()$
\Loop
	\State $S \gets$ \Call{score}{C,E,N,startTime}
	\If{max($S$) $\neq conditionsNb$ \textbf{and} $(time() - startTime) < 2$ hours}
		\State $N \gets$ \Call{evolve}{$N$,$S$}
	\Else
		\State break
	\EndIf
\EndLoop
\EndProcedure
\end{algorithmic}
\end{algorithm}
